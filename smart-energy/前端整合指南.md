    const reader = new FileReader();
    reader.onload = (e) => {
      const lines = e.target.result.split('\n');

          kwh: parseFloat(kwh),
          date: date.trim()
        };
      });
      resolve(records);
    };
    reader.readAsText(file);
  });
}
